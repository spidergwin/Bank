// Create a payment using the NexaPay API
const API_KEY = 'cg_live_your_api_key_here';
const API_URL = 'https://nexapay.one/api/v1';
// Create a payment with webhook callback
async function createPayment() {
  const response = await fetch(`${API_URL}/payments`, {
    method: 'POST',
    headers: {
      'X-API-Key': API_KEY,
      'Content-Type': 'application/json',
    },
    body: JSON.stringify({
      amount: 99.00,
      currency: 'USD',
      crypto: 'USDC',
      description: 'Order #123',
      customer_email: 'customer@example.com',
      success_url: 'https://yoursite.com/success',
      cancel_url: 'https://yoursite.com/cancel',
      callback_url: 'https://yoursite.com/webhooks/nexapay',
    }),
  });
  const data = await response.json();
  // Redirect customer to payment page
  window.location.href = data.payment.checkout_url;
}

Safe Testing

Develop and test without any risk

NexaPay uses a single live API key for both development and production. There is no separate test mode — and you don't need one. Creating a payment session via the API does not charge anyone. A real charge only occurs when a customer completes the full checkout and enters their payment details.
What you can safely test

    Create payment sessions and inspect the API response
    Test the full redirect flow (success_url and cancel_url)
    Verify webhook delivery to your endpoints
    Build and debug your checkout integration end-to-end
    No charges until a real customer completes payment

How it works

When you call the API to create a payment, NexaPay generates a checkout session and returns a checkout URL. No money moves at this point. The customer must visit the checkout page, enter their card details, and confirm the payment for any charge to occur. This means your developers can freely integrate, test redirects, and verify webhooks with zero risk.

Webhooks

Real-time payment notifications

Webhooks allow you to receive real-time notifications on payment events. Provide a callback_url when creating a payment, or set a global URL in your dashboard Settings. NexaPay sends a webhook for each of the following events: payment completed and session expired.
Webhook Security

All webhook requests include HMAC-SHA256 signatures for verification. Configure your webhook secret in the dashboard and verify signatures to prevent spoofing.

X-NexaPay-SignatureHMAC-SHA256 signature

X-NexaPay-TimestampRequest timestamp (for replay protection)

Webhook Payload

{
  "order_id": "live_abc123def456...",
  "payment_id": "12345678",
  "status": "completed",
  "amount": 95.95,
  "txid": "0xabc123...",
  "timestamp": "2026-02-18T01:32:31.881Z"
}

Status Values

pendingPayment session was created. The customer has not yet completed payment.

completedPayment was successfully processed. Fulfil the order.

failedThe payment failed. Notify the customer and allow retry.

expiredThe payment session timed out (24 hours) without a completed payment. Release any held inventory and notify the customer.

Note: amount and txid are present for on-chain payments and may be omitted for card payments.

webhook-handler.jsCopy

// Webhook endpoint example (Node.js/Express)
const crypto = require('crypto');
app.post('/webhooks/nexapay', async (req, res) => {
  const signature = req.headers['x-nexapay-signature'];
  const timestamp = req.headers['x-nexapay-timestamp'];
  const payload = JSON.stringify(req.body);
  // Verify HMAC signature
  const webhookSecret = process.env.NEXAPAY_WEBHOOK_SECRET;
  if (webhookSecret && signature) {
    const expectedSig = 'sha256=' + crypto
      .createHmac('sha256', webhookSecret)
      .update(timestamp + '.' + payload)
      .digest('hex');
    if (signature !== expectedSig) {
      return res.status(401).json({ error: 'Invalid signature' });
    }
    // Reject old webhooks (replay protection)
    const maxAge = 5 * 60 * 1000; // 5 minutes
    if (Math.abs(Date.now() - parseInt(timestamp)) > maxAge) {
      return res.status(401).json({ error: 'Webhook expired' });
    }
  }
  const { order_id, payment_id, status } = req.body;
  if (status === 'completed') {
    await fulfillOrder(order_id);
  } else if (status === 'expired') {
    // Payment session timed out — release held inventory, notify customer
    await cancelOrder(order_id);
  }
  res.json({ received: true });
});
